54  bool IsUnique() const {
55  return GetRefs() == 1;
56  }

58  int32_t GetRefs() const {
59  return mRefs.load(std::memory_order_relaxed);
60  }
61 
62  int32_t AddRef() const {
63  assert(GetRefs() > 0);
64  return mRefs.fetch_add(1);
65  }
66 
67  int32_t RelRef() const {
68  assert(GetRefs() > 0);
69  auto refs = mRefs.fetch_sub(1);
70 
71  if (refs == 1) {
72  // Was last reference
73  // Destroy object and release memory
74  delete this;
75  }
76 
77  return refs;
78  }
79 
80  private:
81  // This type of object after creation always has at least 1 reference
82  mutable std::atomic_int32_t mRefs{1};
83  };
