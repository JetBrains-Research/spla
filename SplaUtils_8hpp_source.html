97  template<typename FileValue>
98  MatrixLoader &Load(std::istream &is) {
99  static_assert(!(std::is_same_v<FileValue, void> && !std::is_same_v<Value, void>) );
100  std::string line;
101  std::size_t lineN = 0;
102  while (std::getline(is, line)) {
103  ++lineN;
104  if (line[0] != '%') break;
105  }
106  Size nnz;
107  std::stringstream headerLineStream(line);
108  headerLineStream >> mNrows >> mNcols >> nnz;
109 
110  if constexpr (HasValue) {
111  mVals.reserve(nnz);
112  }
113  mRows.reserve(nnz);
114  mCols.reserve(nnz);
115 
116  while (std::getline(is, line)) {
117  ++lineN;
118  std::stringstream lineStream(line);
119  Index i, j;
120  lineStream >> i >> j;
121  --i, --j;
122  mRows.push_back(i);
123  mCols.push_back(j);
124 
125  if (!(0 <= i && i < mNrows)) {
126  throw std::logic_error("Row index is out of bounds on the line " + std::to_string(lineN));
127  }
128 
129  if (!(0 <= j && j < mNcols)) {
130  throw std::logic_error("Column index is out of bounds on the line " + std::to_string(lineN));
131  }
132 
133  if constexpr (!std::is_same_v<FileValue, void>) {
134  FileValue value;
135  lineStream >> value;
136  if constexpr (HasValue) {
137  mVals.push_back(static_cast<Value>(value));
138  }
139  }
140  }
141 
142  if (mRows.size() != nnz) {
143  throw std::logic_error("Number of non zero values is not valid");
144  }
145 
146  return *this;
147  }
