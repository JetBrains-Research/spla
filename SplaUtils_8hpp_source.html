132  MatrixLoader &Load(std::istream &is, bool makeUndirected, bool removeSelfLoops, bool ignoreValues, bool verbose = true, const std::string &source = "") {
133  if (verbose) {
134  std::cout << "Loading Matrix-market coordinate format graph...\n";
135  std::cout << " Reading from " << source << "\n";
136 
137  if (removeSelfLoops)
138  std::cout << " Removing self-loops\n";
139  }
140 
141  CpuTimer timer;
142  CpuTimer total;
143  total.Start();
144  timer.Start();
145 
146  std::string line;
147  std::size_t lineN = 0;
148  while (std::getline(is, line)) {
149  ++lineN;
150  if (line[0] != '%') break;
151  }
152  Size nnz;
153  std::stringstream headerLineStream(line);
154  headerLineStream >> mNrows >> mNcols >> nnz;
155 
156  mRows.reserve(nnz);
157  mCols.reserve(nnz);
158  mVals.reserve(nnz);
159 
160  timer.Mark();
161 
162  while (std::getline(is, line)) {
163  ++lineN;
164  std::stringstream lineStream(line);
165  Index i, j;
166  lineStream >> i >> j;
167 
168  if (!(1 <= i && i <= mNrows)) {
169  throw std::logic_error("Row index is out of bounds on the line " + std::to_string(lineN));
170  }
171 
172  if (!(1 <= j && j <= mNcols)) {
173  throw std::logic_error("Column index is out of bounds on the line " + std::to_string(lineN));
174  }
175 
176  if (removeSelfLoops) {
177  if (i == j) {
178  nnz -= 1;
179  continue;
180  }
181  }
182 
183  mRows.push_back(i);
184  mCols.push_back(j);
185 
186  if (!lineStream.eof() && !ignoreValues) {
187  Value v;
188  lineStream >> v;
189  mVals.push_back(v);
190  }
191  }
192 
193  if (ignoreValues)
194  mVals.resize(mRows.size());
195 
196  timer.Stop();
197 
198  if (mRows.size() != nnz) {
199  throw std::logic_error("Number of non zero values is not valid");
200  }
201 
202  if (verbose)
203  std::cout << " Parsing MTX file ("
204  << mNrows << " rows, "
205  << mNcols << " cols, "
206  << nnz << " directed edges)"
207  << " in " << timer.GetElapsedMs() << " ms\n";
208 
209  timer.Start();
210 
211  // Offset indices
212  for (std::size_t k = 0; k < mRows.size(); k++) {
213  mRows[k] -= 1;
214  mCols[k] -= 1;
215  }
216 
217  timer.Stop();
218 
219  if (verbose)
220  std::cout << " Offset indices by -1 in "
221  << timer.GetDurationMs() << " ms\n";
222 
223  if (makeUndirected) {
224  timer.Start();
225  DoubleEdges();
226  timer.Stop();
227 
228  if (verbose)
229  std::cout << " Doubling edges: " << nnz
230  << " to " << GetNvals()
231  << " in " << timer.GetDurationMs() << " ms\n";
232  }
233 
234  {
235  timer.Start();
236  auto oldNnz = GetNvals();
237  SortReduceDuplicates();
238  auto newNnz = GetNvals();
239  timer.Stop();
240 
241  if (verbose)
242  std::cout << " Sort values and reduce duplicates:"
243  << " old " << oldNnz
244  << " new " << newNnz
245  << " diff " << oldNnz - newNnz
246  << " in " << timer.GetDurationMs() << " ms\n";
247  }
248 
249 
250  double averageDegree;
251  std::size_t maxDegree = 0;
252  std::size_t minDegree = 0;
253 
254  if (GetNrows() == GetNcols())
255  ComputeStats(minDegree, maxDegree, averageDegree);
256 
257  total.Stop();
258 
259  if (verbose) {
260  std::cout << " Stats: min.deg " << minDegree
261  << ", max.deg " << maxDegree
262  << ", avg.deg " << averageDegree << "\n";
263  std::cout << " Loaded graph"
264  << " vertices " << GetNrows()
265  << " edges " << GetNvals()
266  << " in " << total.GetElapsedMs() << " ms\n";
267  }
268 
269  return *this;
270  }

318  void SortReduceDuplicates() {
319  struct Entry {
320  Index i;
321  Index j;
322  Value v;
323 
324  bool operator!=(const Entry &e) const {
325  return i != e.i || j != e.j;
326  }
327  };
328  std::vector<Entry> entries(GetNvals());
329 
330  for (std::size_t i = 0; i < GetNvals(); i++)
331  entries[i] = Entry{mRows[i], mCols[i], mVals[i]};
332 
333  std::sort(entries.begin(), entries.end(), [](const Entry &a, const Entry &b) { return a.i < b.i || (a.i == b.i && a.j < b.j); });
334 
335  std::vector<Value> newVals;
336  std::vector<Index> newRows;
337  std::vector<Index> newCols;
338 
339  Entry prev{0xffffffff, 0xffffffff, Value()};
340  for (const auto &e : entries) {
341  if (prev != e) {
342  prev = e;
343  newVals.push_back(e.v);
344  newRows.push_back(e.i);
345  newCols.push_back(e.j);
346  }
347  }
348 
349  std::swap(newVals, mVals);
350  std::swap(newRows, mRows);
351  std::swap(newCols, mCols);
352  }

354  void ComputeStats(std::size_t &minDegree, std::size_t &maxDegree, double &averageDegree) {
355  std::vector<std::size_t> degreePerVertex(GetNrows(), 0);
356 
357  for (std::size_t k = 0; k < GetNvals(); k++) {
358  degreePerVertex[mRows[k]] += 1;
359  }
360 
361  maxDegree = 0;
362  minDegree = GetNvals() + 1;
363  averageDegree = 0.0f;
364 
365  for (auto d : degreePerVertex) {
366  maxDegree = std::max(maxDegree, d);
367  minDegree = std::min(minDegree, d);
368  averageDegree += static_cast<double>(d);
369  }
370 
371  averageDegree = GetNrows() > 0 ? averageDegree / static_cast<double>(GetNrows()) : 0.0;
372  }
