100  template<typename FileValue>
101  MatrixLoader &Load(std::istream &is, bool makeUndirected, bool verbose = true, const std::string &source = "") {
102  CpuTimer loading;
103  loading.Start();
104 
105  static_assert(!(std::is_same_v<FileValue, void> && !std::is_same_v<Value, void>) );
106  std::string line;
107  std::size_t lineN = 0;
108  while (std::getline(is, line)) {
109  ++lineN;
110  if (line[0] != '%') break;
111  }
112  Size nnz;
113  std::stringstream headerLineStream(line);
114  headerLineStream >> mNrows >> mNcols >> nnz;
115 
116  if constexpr (HasValue) {
117  mVals.reserve(nnz);
118  }
119  mRows.reserve(nnz);
120  mCols.reserve(nnz);
121 
122  CpuTimer reading;
123 
124  reading.Start();
125  while (std::getline(is, line)) {
126  ++lineN;
127  std::stringstream lineStream(line);
128  Index i, j;
129  lineStream >> i >> j;
130 
131  if (!(1 <= i && i <= mNrows)) {
132  throw std::logic_error("Row index is out of bounds on the line " + std::to_string(lineN));
133  }
134 
135  if (!(1 <= j && j <= mNcols)) {
136  throw std::logic_error("Column index is out of bounds on the line " + std::to_string(lineN));
137  }
138 
139  mRows.push_back(i);
140  mCols.push_back(j);
141 
142  if constexpr (!std::is_same_v<FileValue, void>) {
143  FileValue value;
144  lineStream >> value;
145  if constexpr (HasValue) {
146  mVals.push_back(static_cast<Value>(value));
147  }
148  }
149  }
150  reading.Stop();
151 
152  if (mRows.size() != nnz) {
153  throw std::logic_error("Number of non zero values is not valid");
154  }
155 
156  // Offset indices
157  for (std::size_t k = 0; k < mRows.size(); k++) {
158  mRows[k] -= 1;
159  mCols[k] -= 1;
160  }
161 
162  CpuTimer doubling;
163 
164  if (makeUndirected) {
165  doubling.Start();
166  DoubleEdges();
167  doubling.Stop();
168  }
169 
170  loading.Stop();
171 
172  if (verbose) {
173  std::cout << "Loading Matrix-market coordinate format graph\n";
174  std::cout << " Reading from " << source << "\n";
175  std::cout << " Parsing MTX file (" << mNrows << " rows, " << mNcols << " cols, " << nnz << " directed edges)"
176  << " in " << reading.GetElapsedMs() << " ms\n";
177  if (makeUndirected) {
178  std::cout << " Doubling edges: " << nnz << " to " << GetNvals()
179  << " in " << doubling.GetElapsedMs() << " ms\n";
180  }
181  std::cout << " Loaded in " << loading.GetElapsedMs() << " ms\n";
182  }
183 
184  return *this;
185  }
