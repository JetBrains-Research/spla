307  compute::vector<value_type> temp_resultValues(count1 + count2, queue.get_context());

315  tile_b.begin(), valueFirst,
316  temp_resultKeys.begin(),
317  temp_resultValues.begin(),
318  counts.begin(),
319  compare, equals);

323

326  compactKey_kernel.tile_size = tile_size;

328  compactKey_kernel.exec(queue);

334  compactValues_kernel.exec(queue);

336  return (counts.end() - 1).read(queue);
