97  << " else a_end = a_mid;\n"
98  << " }\n"
99  << " while(b_start<b_end)\n"
100  << " {\n"
101  << " b_mid = (b_start + b_end)/2;\n"
102  << " if(" << comp(first2[expr<uint_>("b_mid")], expr<value_type>("x")) << ")\n"
103  << " b_start = b_mid+1;\n"
104  << " else b_end = b_mid;\n"
105  << " }\n"
106  << " uint a_run = a_index - a_start;\n"
107  << " uint b_run = b_index - b_start;\n"
108  << " uint x_count = a_run + b_run;\n"
109  << " uint b_advance = max(x_count / 2, x_count - a_run);\n"
110  << " b_end = min(b_count, b_start + b_advance + 1);\n"
111  << " uint temp_start = b_index, temp_end = b_end, temp_mid;"

264  template<
265  typename InputMask,
266  typename InputKey,
267  typename Compare,
268  typename Equals,
269  typename ResizeResult,
270  typename AssignResult>
271  std::size_t MaskByKey(InputMask maskFirst,
272  InputKey keyFirst,
273  Compare compare,
274  Equals equals,
275  ResizeResult resizeResult,
276  AssignResult assignResult,
277  std::size_t maskCount,
278  std::size_t keyCount,
279  boost::compute::command_queue &queue) {
280  using namespace boost;
281  typedef typename std::iterator_traits<InputMask>::value_type key_type;
282 
283  std::size_t tileSize = 1024;
284 
285  compute::vector<compute::uint_> tileA((maskCount + keyCount + tileSize - 1) / tileSize + 1, queue.get_context());
286  compute::vector<compute::uint_> tileB((maskCount + keyCount + tileSize - 1) / tileSize + 1, queue.get_context());
287 
288  // Tile the sets
289  detail::BalancedPathKernel balancedPathKernel;
290  balancedPathKernel.tile_size = tileSize;
291  balancedPathKernel.set_range(maskFirst, maskFirst + maskCount,
292  keyFirst, keyFirst + keyCount,
293  tileA.begin() + 1, tileB.begin() + 1,
294  compare);
295  fill_n(tileA.begin(), 1, 0, queue);
296  fill_n(tileB.begin(), 1, 0, queue);
297  balancedPathKernel.exec(queue);
298 
299  fill_n(tileA.end() - 1, 1, maskCount, queue);
300  fill_n(tileB.end() - 1, 1, keyCount, queue);
301 
302  compute::vector<compute::uint_> counts((maskCount + keyCount + tileSize - 1) / tileSize + 1, queue.get_context());
303  compute::fill_n(counts.end() - 1, 1, 0, queue);
304 
305  // Find result intersections count and offsets to write result of each tile
306  detail::SerialIntersectionCountKernel intersectionCountKernel;
307  intersectionCountKernel.set_range(maskFirst, keyFirst,
308  tileA.begin(), tileA.end(),
309  tileB.begin(),
310  counts.begin(),
311  compare, equals);
312  intersectionCountKernel.exec(queue);
313 
314  // Compute actual counts offsets
315  exclusive_scan(counts.begin(), counts.end(), counts.begin(), queue);
316 
317  // Get result count and resize buffers
318  std::size_t resultCount = (counts.end() - 1).read(queue);
319  resizeResult(resultCount);
320 
321  // Find result intersections
322  detail::SerialIntersectionKernel intersectionKernel;
323  intersectionKernel.set_range(maskFirst, keyFirst, tileA.begin(), tileA.end(),
324  tileB.begin(),
325  counts.begin(),
326  compare, equals, assignResult);
327  intersectionKernel.exec(queue);
328 
329  return resultCount;
330  }
