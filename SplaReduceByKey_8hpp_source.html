52  inline void GenerateUintKeys(const std::vector<std::reference_wrapper<const boost::compute::vector<unsigned int>>> &keysFirst,
53  const boost::compute::vector<unsigned int>::iterator &newKeysFirst,
54  std::size_t preferredWorkGroupSize,
55  boost::compute::command_queue &queue) {
56  using namespace boost;
57  using namespace spla::detail::meta;
58  using compute::uint_;
59 
60  compute::detail::meta_kernel k("spla_reduce_by_key_new_key_flags");
61  const std::size_t count = keysFirst.at(0).get().size();
62  const std::size_t nKeys = keysFirst.size();
63  k.add_set_arg<const uint_>("count", static_cast<uint_>(count));
64 
65  k << k.decl<const uint_>("gid") << " = get_global_id(0);\n"
66  << k.decl<uint_>("value") << " = 0;\n"
67  << "if(gid >= count){\n return;\n}\n"
68  << "if(gid > 0){ \n"
69  << DeclareMultiKey{"key", nKeys} << ";\n"
70  << AssignKey{KeyVar("key"), KeyVec(keysFirst, "gid", k), nKeys} << ";\n"
71  << DeclareMultiKey{"previous_key", nKeys} << ";\n"
72  << AssignKey{KeyVar("previous_key"), KeyVec(keysFirst, "gid - 1", k), nKeys}
73  << " value = " << CompareKey{KeyVar("previous_key"), KeyVar("key"), nKeys} << " ? 0 : 1;\n"
74  << "}\n else {\n"
75  << " value = 0;\n"
76  << "}\n"
77  << newKeysFirst[k.var<const uint_>("gid")] << " = value;\n";
78 
79  const compute::context &context = queue.get_context();
80  compute::kernel kernel = k.compile(context);
81 
82  auto workGroupsNo = static_cast<std::size_t>(
83  std::ceil(static_cast<float>(count) / static_cast<float>(preferredWorkGroupSize)));
84 
85  queue.enqueue_1d_range_kernel(kernel,
86  0,
87  workGroupsNo * preferredWorkGroupSize,
88  preferredWorkGroupSize);
89 
90  inclusive_scan(newKeysFirst, newKeysFirst + static_cast<std::ptrdiff_t>(count),
91  newKeysFirst, queue);

94  inline void CarryOuts(const boost::compute::vector<boost::compute::uint_> &keys,
95  const boost::compute::vector<unsigned char> &values,
96  const boost::compute::vector<boost::compute::uint_>::iterator &carryOutsKeysFirst,
97  const boost::compute::vector<unsigned char>::iterator &carryOutValuesFirst,
98  std::size_t workGroupSize,
99  std::size_t vBytes,
100  const std::string &reduceBody,
101  boost::compute::command_queue &queue) {
102  using namespace boost;
103  using namespace detail::meta;
104  using compute::uint_;
105 
106  compute::vector<uint_>::iterator keysFirst = keys.begin();
107  compute::vector<unsigned char>::iterator valuesFirst = values.begin();
108  compute::detail::meta_kernel k("spla_reduce_by_key_with_scan_carry_outs");
109  k.add_set_arg<const uint_>("count", static_cast<uint_>(keys.size()));
110  size_t localKeysArg = k.add_arg<uint_ *>(compute::memory_object::local_memory, "lkeys");
111  size_t localValsArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "lvals");
112 
113  FunctionApplication reduceOp(k, "spla_reduce", reduceBody, vBytes);
114 
115  k << k.decl<const uint_>("gid") << " = get_global_id(0);\n"
116  << k.decl<const uint_>("wg_size") << " = get_local_size(0);\n"
117  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"
118  << k.decl<const uint_>("group_id") << " = get_group_id(0);\n"
119  << k.decl<uint_>("key") << ";\n"
120  << DeclareVal{"value", vBytes} << ";\n"
121  << "if(gid < count){\n"
122  << k.var<uint_>("key") << " = " << keysFirst[k.var<const uint_>("gid")] << ";\n"
123  << AssignVal{ValVar("value"), ValArrItem(valuesFirst, "gid", vBytes, k), vBytes} << ";\n"
124  << "lkeys[lid] = key;\n"
125  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("value"), vBytes} << ";\n"
126  << "}\n"
127  << DeclareVal{"result", vBytes} << ";\n"
128  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
129  << k.decl<uint_>("other_key") << ";\n"
130  << DeclareVal{"other_value", vBytes} << ";\n"
131  << "for (" << k.decl<uint_>("offset") << " = 1; offset < wg_size; offset *= 2) {\n"
132  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
133  << " if (lid >= offset){\n"
134  << " other_key = lkeys[lid - offset];\n"
135  << " if(other_key == key){\n"
136  << AssignVal{ValVar("other_value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
137  << reduceOp.Apply(ValVar("result"), ValVar("other_value"), ValVar("result")) << ";\n"
138  << " }\n"
139  << " }\n"
140  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
141  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
142  << "}\n"
143  << "if (lid == (wg_size - 1)){\n"
144  << carryOutsKeysFirst[k.var<const uint_>("group_id")] << " = key;\n"
145  << AssignVal{ValArrItem(carryOutValuesFirst, "group_id", vBytes, k), ValVar("result"), vBytes} << ";\n"
146  << "}\n";
147 
148 
149  auto workGroupsNo = static_cast<std::size_t>(
150  std::ceil(static_cast<float>(keys.size()) / static_cast<float>(workGroupSize)));
151 
152  const compute::context &context = queue.get_context();
153  compute::kernel kernel = k.compile(context);
154  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
155  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
156 
157  queue.enqueue_1d_range_kernel(kernel,
158  0,
159  workGroupsNo * workGroupSize,
160  workGroupSize);

163  inline void CarryIns(const boost::compute::vector<boost::compute::uint_>::iterator &carryOutsKeysFirst,
164  const boost::compute::vector<unsigned char>::iterator &carryOutValuesFirst,
165  const boost::compute::vector<unsigned char>::iterator &carryInValuesFirst,
166  std::size_t carryOutSize,
167  std::size_t workGroupSize,
168  std::size_t vBytes,
169  const std::string &reduceBody,
170  boost::compute::command_queue &queue) {
171  using namespace boost;
172  using namespace detail::meta;
173  using compute::uint_;
174 
175  auto valuesPreWorkItem = static_cast<uint_>(
176  std::ceil(static_cast<float>(carryOutSize) / static_cast<float>(workGroupSize)));
177 
178  compute::detail::meta_kernel k("spla_reduce_by_key_with_scan_carry_ins");
179  k.add_set_arg<const uint_>("carry_out_size", uint_(carryOutSize));
180  k.add_set_arg<const uint_>("values_per_work_item", valuesPreWorkItem);
181  std::size_t localKeysArg = k.add_arg<uint_ *>(compute::memory_object::local_memory, "lkeys");
182  std::size_t localValsArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "lvals");
183 
184  FunctionApplication reduceOp(k, "spla_reduce", reduceBody, vBytes);
185 
186  k << k.decl<uint_>("id") << " = get_global_id(0) * values_per_work_item;\n"
187  << k.decl<uint_>("idx") << " = id;\n"
188  << k.decl<const uint_>("wg_size") << " = get_local_size(0);\n"
189  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"
190  << k.decl<const uint_>("group_id") << " = get_group_id(0);\n"
191  << k.decl<uint_>("key") << ";\n"
192  << DeclareVal{"value", vBytes} << ";\n"
193  << k.decl<uint_>("previous_key") << ";\n"
194  << DeclareVal{"result", vBytes} << ";\n"
195  << "if (id < carry_out_size) {\n"
196  << k.var<uint_>("previous_key") << " = " << carryOutsKeysFirst[k.var<const uint_>("id")] << ";\n"
197  << AssignVal{ValVar("result"), ValArrItem(carryOutValuesFirst, "id", vBytes, k), vBytes} << ";\n"
198  << AssignVal{ValArrItem(carryInValuesFirst, "id", vBytes, k), ValVar("result"), vBytes} << ";\n"
199  << "}\n"
200  << k.decl<const uint_>("end") << " = (id + values_per_work_item) <= carry_out_size"
201  << " ? (values_per_work_item + id) : carry_out_size;\n"
202  << "for (idx = idx + 1; idx < end; idx += 1) {\n"
203  << " key = " << carryOutsKeysFirst[k.var<const uint_>("idx")] << ";\n"
204  << AssignVal{ValVar("value"), ValArrItem(carryOutValuesFirst, "idx", vBytes, k), vBytes} << ";\n"
205  << " if(previous_key == key){\n"
206  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("result")) << ";\n"
207  << " }\n else { \n"
208  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
209  << " }\n"
210  << AssignVal{ValArrItem(carryInValuesFirst, "idx", vBytes, k), ValVar("result"), vBytes} << ";\n"
211  << " previous_key = key;\n"
212  "}\n"
213  << "lkeys[lid] = previous_key;\n"
214  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
215  << "for (" << k.decl<uint_>("offset") << " = 1; "
216  << "offset < wg_size; offset *= 2){\n"
217  " barrier(CLK_LOCAL_MEM_FENCE);\n"
218  << " if(lid >= offset){\n"
219  " key = lkeys[lid - offset];\n"
220  << " if(previous_key == key){\n"
221  << AssignVal{ValVar("value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
222  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("result")) << ";\n"
223  << " }\n"
224  << " }\n"
225  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
226  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
227  << "}\n"
228  << "barrier(CLK_LOCAL_MEM_FENCE);\n"
229  << "if(lid > 0){\n"
230  << " previous_key = lkeys[lid - 1];\n"
231  << AssignVal{ValVar("result"), ValArrItem("lvals", "lid - 1", vBytes), vBytes} << ";\n"
232  << "}\n"
233  << "for (idx = id; idx < id + values_per_work_item; idx += 1) {\n"
234  << " barrier( CLK_GLOBAL_MEM_FENCE );\n"
235  << " if (lid > 0 && idx < carry_out_size) {\n"
236  << " key = " << carryOutsKeysFirst[k.var<const uint_>("idx")] << ";\n"
237  << AssignVal{ValVar("value"), ValArrItem(carryInValuesFirst, "idx", vBytes, k), vBytes} << ";\n"
238  << " if(previous_key == key){\n"
239  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("value")) << ";\n"
240  << " }\n"
241  << AssignVal{ValArrItem(carryInValuesFirst, "idx", vBytes, k), ValVar("value"), vBytes} << ";\n"
242  << " }\n"
243  << "}\n";
244 
245  const compute::context &context = queue.get_context();
246  compute::kernel kernel = k.compile(context);
247  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
248  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
249 
250  queue.enqueue_1d_range_kernel(kernel,
251  0,
252  workGroupSize,
253  workGroupSize);
254  }

256  inline void FinalReduction(const std::vector<std::reference_wrapper<const boost::compute::vector<boost::compute::uint_>>> &keys,
257  const boost::compute::vector<unsigned char>::iterator &valuesFirst,
258  const std::vector<std::reference_wrapper<boost::compute::vector<boost::compute::uint_>>> &keysResult,
259  const boost::compute::vector<unsigned char>::iterator &valuesResult,
260  std::size_t count,
261  const boost::compute::vector<boost::compute::uint_>::iterator &newKeysFirst,

355  inline std::size_t ReduceByKeyWithScan(const std::vector<std::reference_wrapper<const boost::compute::vector<boost::compute::uint_>>> &keys,
356  const boost::compute::vector<unsigned char> &values,
357  const std::vector<std::reference_wrapper<boost::compute::vector<boost::compute::uint_>>> &keysResult,
358  boost::compute::vector<unsigned char> &valuesResult,
359  std::size_t valueByteSize,
360  const std::string &reduceBody,
361  boost::compute::command_queue &queue) {
362  using namespace boost;
363  using namespace detail::meta;
364  using compute::uint_;
365 
366  const compute::context &context = queue.get_context();
367  const std::size_t count = keys.at(0).get().size();
368 
369  if (count == 0) {
370  return 0;
371  }
372 
373  const compute::device &device = queue.get_device();
374  const std::size_t workGroupSize = device.get_info<CL_DEVICE_MAX_WORK_GROUP_SIZE>();
375  compute::vector<uint_> newKeys(count, context);
376  GenerateUintKeys(keys, newKeys.begin(), workGroupSize, queue);
377 
378  const auto carryOutSize = static_cast<std::size_t>(
379  std::ceil(static_cast<float>(count) / static_cast<float>(workGroupSize)));
380  compute::vector<uint_> carryOutKeys(carryOutSize, context);
381  compute::vector<unsigned char> carryOutValues(carryOutSize * valueByteSize, context);
382  CarryOuts(newKeys,
383  values,
384  carryOutKeys.begin(),
385  carryOutValues.begin(),
386  workGroupSize,
387  valueByteSize,
388  reduceBody,
389  queue);
390 
391  compute::vector<unsigned char> carryInValues(carryOutSize * valueByteSize, context);
392  CarryIns(carryOutKeys.begin(),
393  carryOutValues.begin(),
394  carryInValues.begin(),
395  carryOutSize,
396  workGroupSize,
397  valueByteSize,
398  reduceBody,
399  queue);
400 
401  const std::size_t resultSize = compute::detail::read_single_value<uint_>(newKeys.get_buffer(), count - 1, queue) + 1;
402 
403  for (compute::vector<uint_> &kRes : keysResult) {
404  kRes.resize(resultSize, queue);
405  }
406  valuesResult.resize(resultSize * valueByteSize, queue);
407 
408  FinalReduction(keys,
409  values.begin(),
410  keysResult,
411  valuesResult.begin(),
412  count,
413  newKeys.begin(),
414  carryOutKeys.begin(),
415  carryInValues.begin(),
416  workGroupSize,
417  valueByteSize,
418  reduceBody,
419  queue);
420 
421  return resultSize;
422  }

532  inline std::size_t ReducePairKey(const boost::compute::vector<unsigned int> &inputIndices1,
533  const boost::compute::vector<unsigned int> &inputIndices2,
534  boost::compute::vector<unsigned int> &outputIndices1,
535  boost::compute::vector<unsigned int> &outputIndices2,
536  boost::compute::command_queue &queue) {
537  using namespace boost;
538 
539  compute::context ctx = queue.get_context();
540  assert(inputIndices1.size() == inputIndices2.size());
541  const std::size_t inputSize = inputIndices1.size();
542 
543  if (inputSize <= 1) {
544  outputIndices1.resize(inputSize, queue);
545  outputIndices2.resize(inputSize, queue);
546  compute::copy(inputIndices1.begin(), inputIndices1.end(), outputIndices1.begin(), queue);
547  compute::copy(inputIndices2.begin(), inputIndices2.end(), outputIndices2.begin(), queue);
548  return inputSize;
549  }
550 
551  compute::vector<unsigned int> outputPos(inputSize, ctx);
552  detail::GenerateUintKeys({std::ref(inputIndices1), std::ref(inputIndices2)},
553  outputPos.begin(),
554  queue.get_device().get_info<CL_DEVICE_MAX_WORK_GROUP_SIZE>(),
555  queue);
556 
557  const std::size_t uniqueNum = (outputPos.end() - 1).read(queue) + 1;
558  outputIndices1.resize(uniqueNum, queue);
559  outputIndices2.resize(uniqueNum, queue);
560 
561  BOOST_COMPUTE_CLOSURE(void, copyToOutput, (unsigned int i), (inputIndices1, inputIndices2, outputIndices1, outputIndices2, outputPos),
562  {
563  if (outputPos[i] != outputPos[i - 1]) {
