323  const VAlignedValue &right,

339  inline void GenerateUintKeys(const std::vector<std::reference_wrapper<const compute::vector<unsigned int>>> &keysFirst,
340  const compute::vector<unsigned int>::iterator &newKeysFirst,
341  std::size_t preferredWorkGroupSize,
342  compute::command_queue &queue) {
343  compute::detail::meta_kernel k("spla_reduce_by_key_new_key_flags");
344  const std::size_t count = keysFirst.at(0).get().size();
345  const std::size_t nKeys = keysFirst.size();
346  k.add_set_arg<const uint_>("count", static_cast<uint_>(count));
347 
348  k << k.decl<const uint_>("gid") << " = get_global_id(0);\n"
349  << k.decl<uint_>("value") << " = 0;\n"
350  << "if(gid >= count){\n return;\n}\n"
351  << "if(gid > 0){ \n"
352  << DeclareMultiKey{"key", nKeys} << ";\n"
353  << AssignKey{KeyVar("key"), KeyVec(keysFirst, "gid", k), nKeys} << ";\n"
354  << DeclareMultiKey{"previous_key", nKeys} << ";\n"
355  << AssignKey{KeyVar("previous_key"), KeyVec(keysFirst, "gid - 1", k), nKeys}
356  << " value = " << CompareKey{KeyVar("previous_key"), KeyVar("key"), nKeys} << " ? 0 : 1;\n"
357  << "}\n else {\n"
358  << " value = 0;\n"
359  << "}\n"
360  << newKeysFirst[k.var<const uint_>("gid")] << " = value;\n";
361 
362  const compute::context &context = queue.get_context();
363  compute::kernel kernel = k.compile(context);
364 
365  auto workGroupsNo = static_cast<std::size_t>(
366  std::ceil(static_cast<float>(count) / static_cast<float>(preferredWorkGroupSize)));
367 
368  queue.enqueue_1d_range_kernel(kernel,
369  0,
370  workGroupsNo * preferredWorkGroupSize,
371  preferredWorkGroupSize);
372 
373  inclusive_scan(newKeysFirst, newKeysFirst + static_cast<std::ptrdiff_t>(count),
374  newKeysFirst, queue);
375  }

377  inline void CarryOuts(const compute::vector<uint_> &keys,
378  const compute::vector<unsigned char> &values,
379  const compute::vector<uint_>::iterator &carryOutsKeysFirst,
380  const compute::vector<unsigned char>::iterator &carryOutValuesFirst,
381  std::size_t workGroupSize,
382  std::size_t vBytes,
383  const std::string &reduceBody,
384  compute::command_queue &queue) {
385  compute::vector<uint_>::iterator keysFirst = keys.begin();
386  compute::vector<unsigned char>::iterator valuesFirst = values.begin();
387  compute::detail::meta_kernel k("spla_reduce_by_key_with_scan_carry_outs");
388  k.add_set_arg<const uint_>("count", static_cast<uint_>(keys.size()));
389  size_t localKeysArg = k.add_arg<uint_ *>(compute::memory_object::local_memory, "lkeys");
390  size_t localValsArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "lvals");
391 
392  ReduceOp reduceOp(k, "spla_reduce", reduceBody, vBytes);
393 
394  k << k.decl<const uint_>("gid") << " = get_global_id(0);\n"
395  << k.decl<const uint_>("wg_size") << " = get_local_size(0);\n"
396  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"
397  << k.decl<const uint_>("group_id") << " = get_group_id(0);\n"
398  << k.decl<uint_>("key") << ";\n"
399  << DeclareVal{"value", vBytes} << ";\n"
400  << "if(gid < count){\n"
401  << k.var<uint_>("key") << " = " << keysFirst[k.var<const uint_>("gid")] << ";\n"
402  << AssignVal{ValVar("value"), ValArrItem(valuesFirst, "gid", vBytes, k), vBytes} << ";\n"
403  << "lkeys[lid] = key;\n"
404  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("value"), vBytes} << ";\n"
405  << "}\n"
406  << DeclareVal{"result", vBytes} << ";\n"
407  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
408  << k.decl<uint_>("other_key") << ";\n"
409  << DeclareVal{"other_value", vBytes} << ";\n"
410  << "for (" << k.decl<uint_>("offset") << " = 1; offset < wg_size; offset *= 2) {\n"
411  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
412  << " if (lid >= offset){\n"
413  << " other_key = lkeys[lid - offset];\n"
414  << " if(other_key == key){\n"
415  << AssignVal{ValVar("other_value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
416  << reduceOp.Apply(ValVar("result"), ValVar("other_value"), ValVar("result")) << ";\n"
417  << " }\n"
418  << " }\n"
419  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
420  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
421  << "}\n"
422  << "if (lid == (wg_size - 1)){\n"
423  << carryOutsKeysFirst[k.var<const uint_>("group_id")] << " = key;\n"
424  << AssignVal{ValArrItem(carryOutValuesFirst, "group_id", vBytes, k), ValVar("result"), vBytes} << ";\n"
425  << "}\n";
426 
427 
428  auto workGroupsNo = static_cast<std::size_t>(
429  std::ceil(static_cast<float>(keys.size()) / static_cast<float>(workGroupSize)));
430 
431  const compute::context &context = queue.get_context();
432  compute::kernel kernel = k.compile(context);
433  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
434  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
435 
436  queue.enqueue_1d_range_kernel(kernel,
437  0,
438  workGroupsNo * workGroupSize,
439  workGroupSize);
440  }

442  inline void CarryIns(const compute::vector<uint_>::iterator &carryOutsKeysFirst,
443  const compute::vector<unsigned char>::iterator &carryOutValuesFirst,
444  const compute::vector<unsigned char>::iterator &carryInValuesFirst,
445  std::size_t carryOutSize,
446  std::size_t workGroupSize,
447  std::size_t vBytes,
448  const std::string &reduceBody,
449  compute::command_queue &queue) {
450  auto valuesPreWorkItem = static_cast<uint_>(
451  std::ceil(static_cast<float>(carryOutSize) / static_cast<float>(workGroupSize)));
452 
453  compute::detail::meta_kernel k("spla_reduce_by_key_with_scan_carry_ins");
454  k.add_set_arg<const uint_>("carry_out_size", uint_(carryOutSize));
455  k.add_set_arg<const uint_>("values_per_work_item", valuesPreWorkItem);
456  std::size_t localKeysArg = k.add_arg<uint_ *>(compute::memory_object::local_memory, "lkeys");
457  std::size_t localValsArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "lvals");
458 
459  ReduceOp reduceOp(k, "spla_reduce", reduceBody, vBytes);
460 
461  k << k.decl<uint_>("id") << " = get_global_id(0) * values_per_work_item;\n"
462  << k.decl<uint_>("idx") << " = id;\n"
463  << k.decl<const uint_>("wg_size") << " = get_local_size(0);\n"
464  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"
465  << k.decl<const uint_>("group_id") << " = get_group_id(0);\n"
466  << k.decl<uint_>("key") << ";\n"
467  << DeclareVal{"value", vBytes} << ";\n"
468  << k.decl<uint_>("previous_key") << ";\n"
469  << DeclareVal{"result", vBytes} << ";\n"
470  << "if (id < carry_out_size) {\n"
471  << k.var<uint_>("previous_key") << " = " << carryOutsKeysFirst[k.var<const uint_>("id")] << ";\n"
472  << AssignVal{ValVar("result"), ValArrItem(carryOutValuesFirst, "id", vBytes, k), vBytes} << ";\n"
473  << AssignVal{ValArrItem(carryInValuesFirst, "id", vBytes, k), ValVar("result"), vBytes} << ";\n"
474  << "}\n"
475  << k.decl<const uint_>("end") << " = (id + values_per_work_item) <= carry_out_size"
476  << " ? (values_per_work_item + id) : carry_out_size;\n"
477  << "for (idx = idx + 1; idx < end; idx += 1) {\n"
478  << " key = " << carryOutsKeysFirst[k.var<const uint_>("idx")] << ";\n"
479  << AssignVal{ValVar("value"), ValArrItem(carryOutValuesFirst, "idx", vBytes, k), vBytes} << ";\n"
480  << " if(previous_key == key){\n"
481  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("result")) << ";\n"
482  << " }\n else { \n"
483  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
484  << " }\n"
485  << AssignVal{ValArrItem(carryInValuesFirst, "idx", vBytes, k), ValVar("result"), vBytes} << ";\n"
486  << " previous_key = key;\n"
487  "}\n"
488  << "lkeys[lid] = previous_key;\n"
489  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
490  << "for (" << k.decl<uint_>("offset") << " = 1; "
491  << "offset < wg_size; offset *= 2){\n"
492  " barrier(CLK_LOCAL_MEM_FENCE);\n"
493  << " if(lid >= offset){\n"
494  " key = lkeys[lid - offset];\n"
495  << " if(previous_key == key){\n"
496  << AssignVal{ValVar("value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
497  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("result")) << ";\n"
498  << " }\n"
499  << " }\n"
500  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
501  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
502  << "}\n"
503  << "barrier(CLK_LOCAL_MEM_FENCE);\n"
504  << "if(lid > 0){\n"
505  << " previous_key = lkeys[lid - 1];\n"
506  << AssignVal{ValVar("result"), ValArrItem("lvals", "lid - 1", vBytes), vBytes} << ";\n"
507  << "}\n"
508  << "for (idx = id; idx < id + values_per_work_item; idx += 1) {\n"
509  << " barrier( CLK_GLOBAL_MEM_FENCE );\n"
510  << " if (lid > 0 && idx < carry_out_size) {\n"
511  << " key = " << carryOutsKeysFirst[k.var<const uint_>("idx")] << ";\n"
512  << AssignVal{ValVar("value"), ValArrItem(carryInValuesFirst, "idx", vBytes, k), vBytes} << ";\n"
513  << " if(previous_key == key){\n"
514  << reduceOp.Apply(ValVar("result"), ValVar("value"), ValVar("value")) << ";\n"
515  << " }\n"
516  << AssignVal{ValArrItem(carryInValuesFirst, "idx", vBytes, k), ValVar("value"), vBytes} << ";\n"
517  << " }\n"
518  << "}\n";
519 
520  const compute::context &context = queue.get_context();
521  compute::kernel kernel = k.compile(context);
522  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
523  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
524 
525  queue.enqueue_1d_range_kernel(kernel,
526  0,
527  workGroupSize,
528  workGroupSize);

553  << k.decl<const uint_>("lid") << " = get_local_id(0);\n"

562  << "}\n"
563  << DeclareVal{"result", vBytes} << ";\n"
564  << AssignVal{ValVar("result"), ValVar("value"), vBytes} << ";\n"
565  << k.decl<uint_>("other_key") << ";\n"
566  << DeclareVal{"other_value", vBytes} << ";\n"
567  << "for (" << k.decl<uint_>("offset") << " = 1; offset < wg_size ; offset *= 2) {\n"
568  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
569  << " if(lid >= offset) {\n"
570  << " other_key = lkeys[lid - offset];\n"
571  << " if(other_key == key){\n"
572  << AssignVal{ValVar("other_value"), ValArrItem("lvals", "lid - offset", vBytes), vBytes} << ";\n"
573  << reduceOp.Apply(ValVar("result"), ValVar("other_value"), ValVar("result")) << ";\n"
574  << " }\n"
575  << " }\n"
576  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
577  << AssignVal{ValArrItem("lvals", "lid", vBytes), ValVar("result"), vBytes} << ";\n"
578  << "}\n"
579  << "if (gid >= count) {\n return;\n};\n"
580  << k.decl<const bool>("save") << " = (gid < (count - 1)) ?"
581  << newKeysFirst[k.var<const uint_>("gid + 1")] << " != key : true;\n"
582  << k.decl<uint_>("carry_in_key") << ";\n"
583  << "if(group_id > 0 && save) {\n"

585  << " if(key == carry_in_key){\n"
586  << AssignVal{ValVar("other_value"), ValArrItem(carryInValuesFirst, "group_id - 1", vBytes, k), vBytes} << ";\n"
587  << reduceOp.Apply(ValVar("result"), ValVar("other_value"), ValVar("result")) << ";\n"
588  << " }\n"
589  << "}\n"
590  << "if (save) {\n"
591  << AssignKey{KeyVec(keysResult, "key", k), KeyVec(keys, "gid", k), nKeys} << ";\n"
592  << AssignVal{ValArrItem(valuesResult, "key", vBytes, k), ValVar("result"), vBytes} << ";\n"
593  << "}\n";
594 
595  auto workGroupsNo = static_cast<std::size_t>(
596  std::ceil(static_cast<float>(count) / static_cast<float>(workGroupSize)));
597 
598  const compute::context &context = queue.get_context();

600  kernel.set_arg(localKeysArg, compute::local_buffer<uint_>(workGroupSize));
601  kernel.set_arg(localValsArg, compute::local_buffer<unsigned char>(workGroupSize * vBytes));
602 
603  queue.enqueue_1d_range_kernel(kernel,
604  0,
605  workGroupsNo * workGroupSize,
606  workGroupSize);
607  }
