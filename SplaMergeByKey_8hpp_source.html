44  class MergeByKeyPathKernel : public boost::compute::detail::meta_kernel {
45  public:
46  unsigned int tileSize;
47 
48  MergeByKeyPathKernel()
49  : meta_kernel("__spla_merge_path"), tileSize(4) {}
50 
51  template<
52  typename OutputIterator1, typename OutputIterator2,
53  typename BinaryIndicesOperator>
54  void SetRange(const std::size_t aCount,
55  const std::size_t bCount,
56  OutputIterator1 resultA,
57  OutputIterator2 resultB,
58  BinaryIndicesOperator isFirstGt) {
59  namespace compute = boost::compute;
60  using compute::uint_;
61 
62  mACount = aCount;
63  mACountArg = add_arg<uint_>("a_count");
64 
65  mBCount = bCount;
66  mBCountArg = add_arg<uint_>("b_count");
67 
68  *this << "uint i = get_global_id(0);\n"
69  << "uint target = (i+1)*" << tileSize << ";\n"
70  << "uint start = max(convert_int(0),convert_int(target)-convert_int(b_count));\n"
71  << "uint end = min(target,a_count);\n"
72  << "uint a_index, b_index;\n"
73  << "while(start<end)\n"
74  << "{\n"
75  << " a_index = (start + end)/2;\n"
76  << " b_index = target - a_index - 1;\n"
77  << " if(!(";
78  isFirstGt(*this, expr<uint_>("a_index"), expr<uint_>("b_index"));
79  *this << "))\n"
80  << " start = a_index + 1;\n"
81  << " else end = a_index;\n"
82  << "}\n"
83  << resultA[expr<uint_>("i")] << " = start;\n"
84  << resultB[expr<uint_>("i")] << " = target - start;\n";
85  }
86 
87  boost::compute::event exec(boost::compute::command_queue &queue) {
88  namespace compute = boost::compute;
89 
90  if ((mACount + mBCount) / tileSize == 0) {
91  return {};
92  }
93 
94  set_arg(mACountArg, compute::uint_(mACount));
95  set_arg(mBCountArg, compute::uint_(mBCount));
96 
97  return exec_1d(queue, 0, (mACount + mBCount) / tileSize);
98  }
99 
100  private:
101  std::size_t mACount{};
102  std::size_t mACountArg{};
103  std::size_t mBCount{};
104  std::size_t mBCountArg{};
105  };

115  template<
116  typename InputIterator1,
117  typename InputIterator2,
118  typename Operator1,
119  typename Operator2,
120  typename Operator3>
121  void SetRange(InputIterator1 tile_first1,
122  InputIterator1 tile_last1,
123  InputIterator2 tile_first2,
124  Operator1 writeIsSecondGt,
125  Operator2 writeAssignResultToFirst,
126  Operator3 writeAssignResultToSecond) {
127  namespace compute = boost::compute;
128  using compute::uint_;
129 
130  mCount = compute::detail::iterator_range_size(tile_first1, tile_last1) - 1;
131 
132  *this << "uint i = get_global_id(0);\n"
133  << "uint start1 = " << tile_first1[expr<uint_>("i")] << ";\n"
134  << "uint end1 = " << tile_first1[expr<uint_>("i+1")] << ";\n"
135  << "uint start2 = " << tile_first2[expr<uint_>("i")] << ";\n"
136  << "uint end2 = " << tile_first2[expr<uint_>("i+1")] << ";\n"
137  << "uint index = i*" << tileSize << ";\n"
138  << "while(start1<end1 && start2<end2)\n"
139  << "{\n"
140  << " if(!(";
141  writeIsSecondGt(*this, expr<uint_>("start1"), expr<uint_>("start2"));
142  *this << "))\n"
143  << " {\n";
144  writeAssignResultToFirst(*this, expr<uint_>("index"), expr<uint_>("start1"));
145  *this << ";\n"
146  << " index++;\n"
147  << " start1++;\n"
148  << " }\n"
149  << " else\n"
150  << " {\n";
151  writeAssignResultToSecond(*this, expr<uint_>("index"), expr<uint_>("start2"));
152  *this << ";\n"
153  << " index++;\n"
154  << " start2++;\n"
155  << " }\n"
156  << "}\n"
157  << "while(start1<end1)\n"
158  << "{\n";
159  writeAssignResultToFirst(*this, expr<uint_>("index"), expr<uint_>("start1"));
160  *this << ";\n"
161  << " index++;\n"
162  << " start1++;\n"
163  << "}\n"
164  << "while(start2<end2)\n"
165  << "{\n";
166  writeAssignResultToSecond(*this, expr<uint_>("index"), expr<uint_>("start2"));
167  *this << ";\n"
168  << " index++;\n"
169  << " start2++;\n"
170  << "}\n";
171  }

176  }
177 
178  return exec_1d(queue, 0, mCount);

185  template<
186  typename Operator1,
187  typename Operator2,
188  typename Operator3>
189  std::ptrdiff_t MergeByKey(
190  const std::size_t count1,
191  const std::size_t count2,
192  Operator1 compareFirstToSecond,
193  Operator2 assignResultToFirst,
194  Operator3 assignResultToSecond,
195  boost::compute::command_queue &queue) {
196  namespace compute = boost::compute;
197  using compute::uint_;
198 
199  const std::size_t tileSize = 1024;
200 
201  compute::vector<uint_> tileA((count1 + count2 + tileSize - 1) / tileSize + 1, queue.get_context());
202  compute::vector<uint_> tileB((count1 + count2 + tileSize - 1) / tileSize + 1, queue.get_context());
203 
204  // Tile the sets
205  MergeByKeyPathKernel tilingKernel;
206  tilingKernel.tileSize = static_cast<unsigned int>(tileSize);
207  tilingKernel.SetRange(count1, count2, tileA.begin() + 1, tileB.begin() + 1, compareFirstToSecond);
208  fill_n(tileA.begin(), 1, uint_(0), queue);
209  fill_n(tileB.begin(), 1, uint_(0), queue);
210  tilingKernel.exec(queue);
211 
212  fill_n(tileA.end() - 1, 1, static_cast<uint_>(count1), queue);
213  fill_n(tileB.end() - 1, 1, static_cast<uint_>(count2), queue);
214 
215  // Merge
216  SerialMergeByKeyKernel mergeKernel;
217  mergeKernel.tileSize = static_cast<unsigned int>(tileSize);
218  mergeKernel.SetRange(tileA.begin(), tileA.end(), tileB.begin(),
219  compareFirstToSecond, assignResultToFirst, assignResultToSecond);
220 
221  mergeKernel.exec(queue);
222 
223  return static_cast<std::ptrdiff_t>(count1 + count2);
224  }

254  void operator()(MetaKernel &k, const MetaIdx &iFst, const MetaIdx &iSnd) {
255  k << aFst[iFst] << " > " << bFst[iSnd] << " || "
256  << "("
257  << aFst[iFst] << " == " << bFst[iSnd] << " && "
258  << aSnd[iFst] << " > " << bSnd[iSnd]
259  << ")";
260  }
