292  inline std::string MakeVisibilityString(Visibility v) {
293  std::string s;
294  switch (v) {

297  break;

339  std::shared_ptr<ReduceApplication> Apply(const VAlignedValue &left,
340  const VAlignedValue &right,
341  const VAlignedValue &result) {
342  std::string lPtr = left.GetPointer();
343  std::string rPtr = right.GetPointer();
344  std::string resPtr = result.GetPointer();
345  if (lPtr != resPtr && rPtr != resPtr) {
346  return std::make_shared<ReduceApplicationRestrict>(mReduceOpName, left, right, result);
347  }
348  return std::make_shared<ReduceApplicationNonRestrict>(mReduceOpName, left, right, result, mVBytes);
349  }
