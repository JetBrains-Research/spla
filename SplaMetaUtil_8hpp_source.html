290  inline std::string PrintVisibility(Visibility v) {
291  std::string s;
292  switch (v) {
293  case (Visibility::Unspecified):
294  s = "";
295  break;
296  case (Visibility::Local):
297  s = "__local";
298  break;
299  case (Visibility::Global):
300  s = "__global";
301  break;
302  default:
303  break;
304  }
305  return s;
306  }

308  inline std::string MakeFunction(const std::string &name, const std::string &body, Visibility accessA, Visibility accessB, Visibility accessC) {
309  std::stringstream fun;
310  fun << "void " << name << " ("
311  << PrintVisibility(accessA) << " const void* vp_a, "
312  << PrintVisibility(accessB) << " const void* vp_b, "
313  << PrintVisibility(accessC) << " void* vp_c) {\n"
314  << "#define _ACCESS_A " << PrintVisibility(accessA) << "\n"
315  << "#define _ACCESS_B " << PrintVisibility(accessB) << "\n"
316  << "#define _ACCESS_C " << PrintVisibility(accessC) << "\n"
317  << body << "\n"
318  << "#undef _ACCESS_A\n"
319  << "#undef _ACCESS_B\n"
320  << "#undef _ACCESS_C\n"
321  << "}";
322  return fun.str();
323  }

344  std::shared_ptr<ReduceApplication> Apply(const VAlignedValue &left,
345  const VAlignedValue &right,
346  const VAlignedValue &result) {
347  std::string lPtr = left.GetPointer();
348  std::string rPtr = right.GetPointer();
349  std::string resPtr = result.GetPointer();
350  if (lPtr != resPtr && rPtr != resPtr) {
351  return std::make_shared<ReduceApplicationRestrict>(mReduceOpName, left, right, result);
352  }
353  assert(mVBytes.has_value() && "To invoke function (non-restrict) value byte size is required");
354  return std::make_shared<ReduceApplicationNonRestrict>(mReduceOpName, left, right, result, mVBytes.value());
355  }
356 
357  private:
358  explicit FunctionApplication(MetaKernel &k,
359  std::string reduceOpName,
360  const std::string &body,
361  std::optional<std::size_t> vBytes,
362  Visibility aVisibility,
363  Visibility bVisibility,
364  Visibility cVisibility)
365  : mReduceOpName(std::move(reduceOpName)),
366  mVBytes(vBytes) {
367 
368  std::stringstream splaReduceOp;
369 
370  std::string aAccess = PrintVisibility(aVisibility);
371  std::string bAccess = PrintVisibility(bVisibility);
372  std::string cAccess = PrintVisibility(cVisibility);
373 
374  splaReduceOp << MakeFunction(mReduceOpName,
375  body,
376  aVisibility,
377  bVisibility,
378  cVisibility);
379 
380  k.add_function(mReduceOpName, splaReduceOp.str());
381  }
