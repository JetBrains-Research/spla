39  inline void InplaceReduce(const boost::compute::vector<unsigned char> &values,
40  std::size_t valueByteSize,
41  const std::string &functionBody,
42  boost::compute::command_queue &queue) {
43  std::size_t inputSize = values.size() / valueByteSize;
44  if (inputSize < 2) {
45  return;
46  }
47 
48  const compute::context &context = queue.get_context();
49 
50  const std::size_t blockSize = 64;
51  const std::size_t valuesPerThread = 8;
52  std::size_t blockCount = inputSize / (blockSize * valuesPerThread);
53  if (blockCount * blockSize * valuesPerThread != inputSize)
54  blockCount++;
55 
56  compute::vector<unsigned char> output(blockCount * valueByteSize, context);
57 
58  compute::detail::meta_kernel k("inplace_reduce");
59  size_t inputArg = k.add_arg<unsigned char *>(compute::memory_object::global_memory, "input");
60  size_t inputSizeArg = k.add_arg<const uint_>("input_size");
61  size_t outputArg = k.add_arg<unsigned char *>(compute::memory_object::global_memory, "output");
62  size_t scratchArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "scratch");
63 
64  ReduceOp reduceOp1(k, "inplace_reduce_reduce_1", functionBody, valueByteSize,
65  Visibility::Unspecified,
66  Visibility::Global,
67  Visibility::Unspecified);
68 
69  ReduceOp reduceOp2(k, "inplace_reduce_reduce_2", functionBody, valueByteSize,
70  Visibility::Local,
71  Visibility::Local,
72  Visibility::Unspecified);
73 
74  k << "const uint gid = get_global_id(0);\n"
75  << "const uint lid = get_local_id(0);\n"
76  << "const uint values_per_thread = " << uint_(valuesPerThread) << ";\n"
77  << "const uint index = gid * values_per_thread;\n"
78  << "if(index < input_size){\n"
79  << DeclareVal{"sum", valueByteSize} << ";\n"
80  << AssignVal{ValVar{"sum"}, ValArrItem{"input", "index", valueByteSize}, valueByteSize} << ";\n"
81  << "for(uint i = 1; i < values_per_thread && (index + i) < input_size; i++) {\n"
82  << reduceOp1.Apply(ValVar{"sum"}, ValArrItem{"input", "index+i", valueByteSize}, ValVar{"sum"}) << ";\n"
83  << "}\n"
84  << AssignVal{ValArrItem("scratch", "lid", valueByteSize), ValVar("sum"), valueByteSize} << ";\n"
85  << "}\n"
86  << "for(uint i = 1; i < get_local_size(0); i <<= 1){\n"
87  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
88  << " uint mask = (i << 1) - 1;\n"
89  << " uint next_index = (gid + i) * values_per_thread;\n"
90  " if((lid & mask) == 0 && next_index < input_size){\n"
91  << reduceOp2.Apply(ValArrItem("scratch", "lid", valueByteSize), ValArrItem("scratch", "lid+i", valueByteSize), ValArrItem("scratch", "lid", valueByteSize))
92  << " }\n"
93  << "}\n"
94  << "if(lid == 0){\n"
95  << AssignVal{ValArrItem("output", "get_group_id(0)", valueByteSize), ValArrItem("scratch", "0", valueByteSize), valueByteSize} << ";\n"
96  << "}\n";
97 
98  const compute::buffer *inputBuffer = &values.get_buffer();
99  const compute::buffer *outputBuffer = &output.get_buffer();
100 
101  compute::kernel kernel = k.compile(context);
102 
103  while (inputSize > 1) {
104  kernel.set_arg(inputArg, *inputBuffer);
105  kernel.set_arg(inputSizeArg, static_cast<uint_>(inputSize));
106  kernel.set_arg(outputArg, *outputBuffer);
107  kernel.set_arg(scratchArg, compute::local_buffer<unsigned char>(blockSize * valueByteSize));
108 
109  queue.enqueue_1d_range_kernel(kernel,
110  0,
111  blockCount * blockSize,
112  blockSize);
113 
114  inputSize = static_cast<std::size_t>(std::ceil(static_cast<float>(inputSize) / (blockSize * valuesPerThread)));
115 
116  blockCount = inputSize / (blockSize * valuesPerThread);
117  if (blockCount * blockSize * valuesPerThread != inputSize)
118  blockCount++;
119 
120  std::swap(inputBuffer, outputBuffer);
121  }
122 
123  if (inputBuffer != &values.get_buffer()) {
124  ::boost::compute::copy(output.begin(),
125  output.begin() + static_cast<std::ptrdiff_t>(valueByteSize),
126  values.begin(),
127  queue);
128  }
129  }

131  inline std::size_t Reduce(const boost::compute::vector<unsigned char> &values,
132  std::size_t valueByteSize,
133  boost::compute::vector<unsigned char> &result,
134  std::size_t blockSize,
135  const std::string &functionBody,
136  boost::compute::command_queue &queue) {
137  using namespace boost;
138 
139  const compute::context &context = queue.get_context();
140  const std::size_t nValues = values.size() / valueByteSize;
141  const std::size_t blockCount = nValues / 2 / blockSize;
142  const auto totalBlockCount = static_cast<std::size_t>(
143  std::ceil(static_cast<float>(nValues) / 2.f / static_cast<float>(blockSize)));
144 
145  if (blockCount != 0) {
146  compute::detail::meta_kernel k("block_reduce");
147  std::size_t outputArg = k.add_arg<unsigned char *>(compute::memory_object::global_memory, "output");
148  std::size_t blockArg = k.add_arg<unsigned char *>(compute::memory_object::local_memory, "block");
149 
150  ReduceOp reduceOpGlobal(k, "block_reduce_reduce_global", functionBody, valueByteSize,
151  Visibility::Global,
152  Visibility::Global,
153  Visibility::Local);
154  ReduceOp reduceOpLocal(k, "block_reduce_reduce_local", functionBody, valueByteSize,
155  Visibility::Local,
156  Visibility::Local,
157  Visibility::Unspecified);
158 
159  k << "const uint gid = get_global_id(0);\n"
160  << "const uint lid = get_local_id(0);\n"
161  << reduceOpGlobal.Apply(ValArrItem(values, "gid*2+0", valueByteSize, k),
162  ValArrItem(values, "gid*2+1", valueByteSize, k),
163  ValArrItem("block", "lid", valueByteSize))
164  << "for(uint i = 1; i < " << uint_(blockSize) << "; i <<= 1){\n"
165  << " barrier(CLK_LOCAL_MEM_FENCE);\n"
166  << " uint mask = (i << 1) - 1;\n"
167  << " if((lid & mask) == 0){\n"
168  << reduceOpLocal.Apply(ValArrItem("block", "lid", valueByteSize),
169  ValArrItem("block", "lid+i", valueByteSize),
170  ValArrItem("block", "lid", valueByteSize))
171  << " }\n"
172  << "}\n"
173  << "if(lid == 0)\n"
174  << AssignVal{ValArrItem{"output", "get_group_id(0)", valueByteSize},
175  ValArrItem{"block", "0", valueByteSize},
176  valueByteSize}
177  << ";\n";
178 
179  compute::kernel kernel = k.compile(context);
180  kernel.set_arg(outputArg, result.get_buffer());
181  kernel.set_arg(blockArg, compute::local_buffer<unsigned char>(blockSize * valueByteSize));
182 
183  queue.enqueue_1d_range_kernel(kernel,
184  0,
185  blockCount * blockSize,
186  blockSize);
187  }
188 
189  // serially reduce any leftovers
190  if (blockCount * blockSize * 2 < nValues) {
191  std::size_t lastBlockStart = blockCount * blockSize * 2;
192 
193  compute::detail::meta_kernel k("extra_serial_reduce");
194  const std::size_t countArg = k.add_arg<uint_>("count");
195  const std::size_t offsetArg = k.add_arg<uint_>("offset");
196  const std::size_t outputArg = k.add_arg<unsigned char *>(compute::memory_object::global_memory, "output");
197  const std::size_t outputOffsetArg = k.add_arg<uint_>("output_offset");
198 
199  ReduceOp reduceOp(k, "leftover_reduce_reduce", functionBody, valueByteSize,
200  Visibility::Unspecified,
201  Visibility::Global,
202  Visibility::Unspecified);
203 
204  k << DeclareVal{"result", valueByteSize} << ";\n"
205  << AssignVal{ValVar{"result"}, ValArrItem{values, "offset", valueByteSize, k}, valueByteSize}
206  << "for(uint i = offset + 1; i < count; i++)\n"
207  << reduceOp.Apply(ValVar{"result"},
208  ValArrItem(values, "i", valueByteSize, k),
209  ValVar{"result"})
210  << AssignVal{ValArrItem{"output", "output_offset", valueByteSize},
211  ValVar{"result"},
212  valueByteSize}
213  << ";\n";
214 
215  compute::kernel kernel = k.compile(context);
216  kernel.set_arg(countArg, static_cast<uint_>(nValues));
217  kernel.set_arg(offsetArg, static_cast<uint_>(lastBlockStart));
218  kernel.set_arg(outputArg, result.get_buffer());
219  kernel.set_arg(outputOffsetArg, static_cast<uint_>(blockCount));
220 
221  queue.enqueue_task(kernel);
222  }
223 
224  return totalBlockCount;
225  }

228  std::size_t valueByteSize,

246  inline void GenericReduce(const boost::compute::vector<unsigned char> &values,
247  std::size_t valueByteSize,
248  boost::compute::vector<unsigned char> &result,
249  const std::string &functionBody,
250  boost::compute::command_queue &queue) {

252

254 
255  compute::vector<unsigned char> results = BlockReduce(values, valueByteSize, blockSize, functionBody, queue);
256
